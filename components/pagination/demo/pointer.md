基本的分页器。

````jsx
import { Pagination, WhiteSpace, WingBlank } from 'antFB-mobile';

const App = React.createClass({
  render() {
    return (
      <div>
        <div className="pagination-container" >
          <WhiteSpace size="lg" />
          <WingBlank>
            <p className="title">点状</p>
            <div style={{ marginBottom: '32px' }}>
              <Pagination
                mode="pointer"
                total={5}
                current={2}
              />
            </div>
          </WingBlank>
          <WhiteSpace size="lg" />
        </div>
      </div>
    );
  },
});

ReactDOM.render(<App />, mountNode);
````
